// --------- SIDEBAR TOGGLE ---------
function setupSidebar() {
    const sidebar = document.getElementById("sidebar");
    const mainCards = document.querySelectorAll(".main-card");
    const toggleBtn = document.getElementById("sidebar-toggle");
    
    toggleBtn.addEventListener("click", () => {
        sidebar.classList.toggle("collapsed");
        mainCards.forEach(card => {
            card.classList.toggle("sidebar-collapsed");
        });
    });
}

// --------- SIDEBAR NAVIGATION ---------
function setupNavigation() {
    const sidebarItems = document.querySelectorAll(".sidebar-item");
    
    sidebarItems.forEach(item => {
        item.addEventListener("click", (e) => {
            e.preventDefault();
            
            // Remove active from all
            sidebarItems.forEach(i => i.classList.remove("active"));
            item.classList.add("active");
            
            // Hide all sections
            document.querySelectorAll(".section-page").forEach(section => {
                section.style.display = "none";
            });
            
            // Show selected section
            const sectionId = item.getAttribute("data-section");
            const targetSection = document.getElementById(sectionId);
            if (targetSection) {
                targetSection.style.display = "flex";
            }
        });
    });
}

// --------- MACRO STATE ---------
let macroUpdateInterval = null;

// --------- UPDATE MACRO STATUS ---------
async function updateMacroStatus() {
    try {
        const status = await window.pywebview.api.macro_get_status();
        
        // Update stat cards
        document.querySelector('.stat-card:nth-child(1) .stat-value').textContent = status.active_clickers;
        document.querySelector('.stat-card:nth-child(2) .stat-value').textContent = status.total_clickers;
        
        const statusText = document.querySelector('.stat-card:nth-child(3) .stat-value');
        if (status.is_clicking) {
            statusText.textContent = 'Clicking';
            statusText.style.color = '#6ef5b8';
        } else if (status.is_running) {
            statusText.textContent = 'Active';
            statusText.style.color = '#ffb300';
        } else {
            statusText.textContent = 'Idle';
            statusText.style.color = '#8d8d8d';
        }
        
        // Update status dots
        const dot1 = document.querySelector('.stat-card:nth-child(1) .status-dot');
        const dot2 = document.querySelector('.stat-card:nth-child(2) .status-dot');
        
        if (status.active_clickers > 0) {
            dot1.classList.add('status-active');
        } else {
            dot1.classList.remove('status-active');
        }
        
        if (status.is_running) {
            dot2.classList.add('status-active');
        } else {
            dot2.classList.remove('status-active');
        }
        
        // Update keybind display
        const keybindDisplay = document.getElementById('keybind-display');
        if (keybindDisplay && status.hotkey) {
            keybindDisplay.textContent = status.hotkey;
        }
        
        // Update CPS input
        const cpsInput = document.getElementById('cps-input');
        if (cpsInput && status.cps > 0) {
            cpsInput.value = Math.round(status.cps);
        }
        
        // Update mode select
        const modeSelect = document.querySelector('.control-select');
        if (modeSelect && status.mode) {
            modeSelect.value = status.mode.charAt(0).toUpperCase() + status.mode.slice(1);
        }
        
    } catch (err) {
        console.error("Error updating macro status:", err);
    }
}

// --------- CLASH MACRO SETUP ---------
function setupClashMacro() {
    // CPS input handler
    const cpsInput = document.getElementById('cps-input');
    if (cpsInput) {
        cpsInput.addEventListener('input', async () => {
            const cps = parseInt(cpsInput.value) || 10;
            
            if (cps < 1 || cps > 1000) {
                return;
            }
            
            try {
                const result = await window.pywebview.api.macro_set_cps(cps);
                if (result.success) {
                    console.log('CPS set to:', cps);
                }
            } catch (err) {
                console.error("Error setting CPS:", err);
            }
        });
    }
    
    // Mode select handler
    const modeSelect = document.querySelector('#clash-macro .control-select');
    if (modeSelect) {
        modeSelect.addEventListener('change', async () => {
            const mode = modeSelect.value.toLowerCase();
            console.log('Changing mode to:', mode);
            
            try {
                const result = await window.pywebview.api.macro_set_mode(mode);
                if (result.success) {
                    showNotification('success', 'Mode Changed', result.message);
                }
            } catch (err) {
                console.error("Error setting mode:", err);
            }
        });
    }
    
    // Start button handler
    const startBtn = document.getElementById('start-macro-btn');
    if (startBtn) {
        startBtn.addEventListener('click', async () => {
            try {
                const status = await window.pywebview.api.macro_get_status();
                
                if (status.is_running) {
                    const result = await window.pywebview.api.macro_stop();
                    if (result.success) {
                        startBtn.innerHTML = `
                            <svg width="16" height="16" viewBox="0 0 24 24" fill="none" xmlns="http://www.w3.org/2000/svg">
                                <path d="M5 3L19 12L5 21V3Z" fill="currentColor"/>
                            </svg>
                            Start
                        `;
                        showNotification('warning', 'Macro Stopped', result.message);
                    }
                } else {
                    const result = await window.pywebview.api.macro_start();
                    if (result.success) {
                        startBtn.innerHTML = `
                            <svg width="16" height="16" viewBox="0 0 24 24" fill="none" xmlns="http://www.w3.org/2000/svg">
                                <rect x="6" y="4" width="4" height="16" fill="currentColor"/>
                                <rect x="14" y="4" width="4" height="16" fill="currentColor"/>
                            </svg>
                            Stop
                        `;
                        showNotification('success', 'Macro Started', result.message);
                    } else {
                        showNotification('error', 'Start Failed', result.message);
                    }
                }
                
                await updateMacroStatus();
            } catch (err) {
                console.error("Error toggling macro:", err);
            }
        });
    }
    
    // Set Hotkey button
    const setHotkeyBtn = document.getElementById('set-hotkey-btn');
    if (setHotkeyBtn) {
        setHotkeyBtn.addEventListener('click', showSetHotkeyModal);
    }
    
    // Status updates
    const macroSection = document.getElementById('clash-macro');
    if (macroSection) {
        const observer = new MutationObserver((mutations) => {
            mutations.forEach((mutation) => {
                if (mutation.attributeName === 'style') {
                    if (macroSection.style.display === 'flex') {
                        updateMacroStatus();
                        if (!macroUpdateInterval) {
                            macroUpdateInterval = setInterval(updateMacroStatus, 500);
                        }
                    } else {
                        if (macroUpdateInterval) {
                            clearInterval(macroUpdateInterval);
                            macroUpdateInterval = null;
                        }
                    }
                }
            });
        });
        
        observer.observe(macroSection, { attributes: true });
    }
}

// --------- LOAD FLAGS FROM PYTHON ---------
async function loadFlags() {
    try {
        const flags = await window.pywebview.api.get_flags();
        const tbody = document.querySelector("#flags-table tbody");
        const totalSpan = document.getElementById("total-flags");

        tbody.innerHTML = "";
        flags.forEach(([name, value]) => {
            const tr = document.createElement("tr");
            tr.innerHTML = `
                <td class="flag-name">${name}</td>
                <td contenteditable="true" class="flag-value">${value}</td>
            `;
            
            // Make rows selectable
            tr.addEventListener("click", (e) => {
                // Don't toggle if clicking on editable cell
                if (e.target.contentEditable === "true") return;
                tr.classList.toggle("selected");
            });
            
            // Save on value edit
            const valueCell = tr.querySelector(".flag-value");
            valueCell.addEventListener("blur", async () => {
                const newValue = valueCell.textContent.trim();
                const flagName = tr.querySelector(".flag-name").textContent;
                
                try {
                    await window.pywebview.api.update_flag(flagName, newValue);
                    showNotification('success', 'Flag Updated', `${flagName} = ${newValue}`);
                } catch (err) {
                    console.error("Error updating flag:", err);
                    showNotification('error', 'Update Failed', 'Failed to update flag');
                    await loadFlags(); // Reload to revert
                }
            });
            
            // Prevent newlines in contenteditable
            valueCell.addEventListener("keydown", (e) => {
                if (e.key === "Enter") {
                    e.preventDefault();
                    valueCell.blur();
                }
            });
            
            tbody.appendChild(tr);
        });

        totalSpan.textContent = flags.length.toString();
    } catch (err) {
        console.error("Error loading flags:", err);
    }
}

// --------- SEARCH FILTER (FIXED & ENHANCED) ---------
function setupSearch() {
    const searchInput = document.getElementById("search");
    const tbody = document.querySelector("#flags-table tbody");
    const totalSpan = document.getElementById("total-flags");

    if (!searchInput) {
        console.error("Search input not found!");
        return;
    }

    if (!tbody) {
        console.error("Table body not found!");
        return;
    }

    console.log("Search initialized successfully");

    searchInput.addEventListener("input", (e) => {
        const query = searchInput.value.toLowerCase().trim();
        const rows = tbody.querySelectorAll("tr");
        let visibleCount = 0;

        console.log("Searching for:", query);

        rows.forEach(row => {
            const nameCell = row.querySelector(".flag-name");
            const valueCell = row.querySelector(".flag-value");
            
            if (!nameCell) return;
            
            const name = nameCell.textContent.toLowerCase();
            const value = valueCell ? valueCell.textContent.toLowerCase() : "";
            
            // Check if query matches name OR value
            const matches = name.includes(query) || value.includes(query);
            
            if (matches || query === "") {
                row.style.display = "";
                visibleCount++;
            } else {
                row.style.display = "none";
            }
        });

        // Update the counter
        const totalFlags = rows.length;
        if (query) {
            totalSpan.textContent = `${visibleCount} / ${totalFlags}`;
            totalSpan.style.color = visibleCount === 0 ? "#f56e6e" : "#ffb300";
        } else {
            totalSpan.textContent = totalFlags.toString();
            totalSpan.style.color = "#9a9a9a";
        }

        console.log(`Found ${visibleCount} matching flags out of ${totalFlags}`);
    });
}

// --------- WINDOW CONTROL BUTTONS (CLEAN VERSION) ---------
function setupWindowButtons() {
    const closeBtns = document.querySelectorAll(".close-btn");
    const minBtns = document.querySelectorAll(".min-btn");
    const appContainer = document.querySelector(".app-container");

    // CLOSE BUTTONS - Closes the app immediately
    closeBtns.forEach((btn) => {
        btn.onclick = function(e) {
            e.stopPropagation();
            window.pywebview.api.close();
        };
    });

    // MINIMIZE BUTTONS - Minimizes with animation
    minBtns.forEach((btn) => {
        btn.onclick = async function(e) {
            e.stopPropagation();
            
            // Add animation
            if (appContainer) {
                appContainer.classList.add('minimizing');
            }
            
            // Wait for animation (250ms)
            await new Promise(resolve => setTimeout(resolve, 250));
            
            // Then minimize
            await window.pywebview.api.minimize();
            
            // Cleanup
            setTimeout(() => {
                if (appContainer) {
                    appContainer.classList.remove('minimizing');
                }
            }, 100);
        };
    });
}

// --------- MODAL UTILITIES ---------
function createModal(title, content) {
    const overlay = document.createElement("div");
    overlay.className = "modal-overlay";
    
    const modal = document.createElement("div");
    modal.className = "modal";
    
    modal.innerHTML = `
        <div class="modal-header">
            <h2 class="modal-title">${title}</h2>
            <button class="modal-close">Ã—</button>
        </div>
        <div class="modal-content">
            ${content}
        </div>
    `;
    
    overlay.appendChild(modal);
    document.body.appendChild(overlay);
    
    modal.querySelector(".modal-close").addEventListener("click", () => {
        overlay.remove();
    });
    
    overlay.addEventListener("click", (e) => {
        if (e.target === overlay) {
            overlay.remove();
        }
    });
    
    return { overlay, modal };
}

// --------- CLEAN INVALID FLAGS ---------
async function cleanInvalidFlags() {
    try {
        const result = await window.pywebview.api.validate_and_clean_flags();
        
        if (result.success) {
            if (result.removed_count > 0) {
                showNotification('warning', 'Flags Cleaned', 
                    `Removed ${result.removed_count} invalid flag(s). ${result.remaining_count} flags remaining.`);
                await loadFlags(); // Refresh the table
            } else {
                showNotification('success', 'All Valid', 'All flags are valid!');
            }
        } else {
            showNotification('error', 'Clean Failed', result.message);
        }
    } catch (err) {
        console.error("Error cleaning flags:", err);
        showNotification('error', 'Error', 'Failed to clean flags');
    }
}

// --------- ADD NEW MODAL ---------
function showAddNewModal() {
    const content = `
        <div class="tabs">
            <button class="tab-btn active" data-tab="single">Add single</button>
            <button class="tab-btn" data-tab="import">Import JSON</button>
        </div>
        
        <div class="tab-content active" id="single-tab">
            <div class="form-group">
                <label class="form-label">Name</label>
                <input type="text" class="modal-input" id="flag-name-input" />
            </div>
            <div class="form-group">
                <label class="form-label">Value</label>
                <input type="text" class="modal-input" id="flag-value-input" placeholder="True/False or number" />
            </div>
        </div>
        
        <div class="tab-content" id="import-tab">
            <div class="form-group">
                <label class="form-label">Paste JSON</label>
                <textarea class="modal-textarea" id="json-import-input" placeholder='{"FlagName": "value", "AnotherFlag": "123"}'></textarea>
            </div>
        </div>
        
        <div class="modal-footer">
            <button class="btn btn-primary" id="add-ok-btn">OK</button>
            <button class="btn btn-ghost" id="add-cancel-btn">Cancel</button>
        </div>
    `;
    
    const { overlay, modal } = createModal("Add Fast Flag", content);
    
    modal.querySelectorAll(".tab-btn").forEach(btn => {
        btn.addEventListener("click", () => {
            modal.querySelectorAll(".tab-btn").forEach(b => b.classList.remove("active"));
            modal.querySelectorAll(".tab-content").forEach(c => c.classList.remove("active"));
            
            btn.classList.add("active");
            const tabId = btn.dataset.tab + "-tab";
            modal.querySelector("#" + tabId).classList.add("active");
        });
    });
    
    modal.querySelector("#add-ok-btn").addEventListener("click", async () => {
        const activeTab = modal.querySelector(".tab-btn.active").dataset.tab;
        
        if (activeTab === "single") {
            const name = modal.querySelector("#flag-name-input").value.trim();
            const value = modal.querySelector("#flag-value-input").value.trim();
            
            if (name && value) {
                try {
                    await window.pywebview.api.add_flag(name, value);
                    await loadFlags();
                    showNotification('success', 'Flag Added', `Successfully added ${name}`);
                    overlay.remove();
                } catch (err) {
                    console.error("Error adding flag:", err);
                    showNotification('error', 'Add Failed', 'Failed to add flag');
                }
            } else {
                showNotification('error', 'Invalid Input', 'Please fill in both name and value');
            }
        } else {
            // FIXED IMPORT JSON SECTION
            const json = modal.querySelector("#json-import-input").value;
            
            if (json) {
                try {
                    JSON.parse(json); // Validate JSON first
                    const result = await window.pywebview.api.import_json(json);
                    
                    if (result && result.success) {
                        await loadFlags();
                        
                        // Show detailed import result
                        const removedCount = result.validation?.removed_count || 0;
                        let message = `Imported ${result.imported} flag(s)`;
                        
                        if (removedCount > 0) {
                            message += `, removed ${removedCount} invalid flag(s)`;
                            showNotification('warning', 'Flags Imported', message);
                        } else {
                            showNotification('success', 'Flags Imported', message);
                        }
                        overlay.remove();
                    } else {
                        showNotification('error', 'Import Failed', result.message || 'Failed to import flags');
                    }
                } catch (e) {
                    showNotification('error', 'Invalid JSON', 'Please enter valid JSON format');
                }
            } else {
                showNotification('error', 'No JSON', 'Please paste JSON data to import');
            }
        }
    });
    
    modal.querySelector("#add-cancel-btn").addEventListener("click", () => {
        overlay.remove();
    });
}

                    Press any key or mouse button (including side buttons)
                </p>
            </div>
            
            <div class="hotkey-display-box">
                <div class="hotkey-key" id="hotkey-display-key">None</div>
                <div class="hotkey-hint">Waiting for input...</div>
            </div>
        </div>
        
        <div class="modal-footer">
            <button class="btn btn-primary" id="hotkey-save-btn">Save</button>
            <button class="btn btn-ghost" id="hotkey-cancel-btn">Cancel</button>
        </div>
    `;
    
    const { overlay, modal } = createModal("Set Hotkey", content);
    
    let capturedKey = null;
    let keyType = null;
    const displayKey = modal.querySelector("#hotkey-display-key");
    const hint = modal.querySelector(".hotkey-hint");
    let isCapturing = true;
    
    // Disable dragging
    document.body.style.webkitAppRegion = 'no-drag';
    overlay.style.webkitAppRegion = 'no-drag';
    modal.style.webkitAppRegion = 'no-drag';
    
    // Capture keyboard
    const handleKeyPress = (e) => {
        if (!isCapturing) return;
        
        let key = e.key.length === 1 ? e.key.toUpperCase() : e.key;
        
        if (['Meta'].includes(key)) return;
        
        e.preventDefault();
        e.stopPropagation();
        
        capturedKey = key;
        keyType = "keyboard";
        displayKey.textContent = key;
        displayKey.style.color = "#ffb300";
        hint.textContent = "Key captured! Click Save to confirm.";
        hint.style.color = "#6ef5b8";
    };
    
    // Capture mouse
    const handleMouseDown = (e) => {
        if (!isCapturing) return;
        
        // Allow clicks on modal buttons
        if (e.target.closest('.btn')) {
            return;
        }
        
        e.preventDefault();
        e.stopPropagation();
        
        let buttonName = "";
        let buttonValue = "";
        
        switch(e.button) {
            case 0: 
                buttonName = "Left Mouse";
                buttonValue = "left";
                break;
            case 1: 
                buttonName = "Middle Mouse";
                buttonValue = "middle";
                break;
            case 2: 
                buttonName = "Right Mouse";
                buttonValue = "right";
                break;
            case 3: 
                buttonName = "Mouse 4 (Back)";
                buttonValue = "x1";
                break;
            case 4: 
                buttonName = "Mouse 5 (Forward)";
                buttonValue = "x2";
                break;
            default: 
                buttonName = `Mouse ${e.button}`;
                buttonValue = `button${e.button}`;
        }
        
        capturedKey = buttonValue;
        keyType = "mouse";
        displayKey.textContent = buttonName;
        displayKey.style.color = "#ffb300";
        hint.textContent = "Mouse button captured! Click Save to confirm.";
        hint.style.color = "#6ef5b8";
    };
    
    const preventContext = (e) => {
        if (isCapturing) {
            e.preventDefault();
        }
    };
    
    const preventNav = (e) => {
        if (isCapturing && (e.button === 3 || e.button === 4)) {
            e.preventDefault();
            return false;
        }
    };
    
    document.addEventListener("keydown", handleKeyPress, true);
    document.addEventListener("mousedown", handleMouseDown, true);
    document.addEventListener("contextmenu", preventContext, true);
    document.addEventListener("mouseup", preventNav, true);
    
    const cleanup = () => {
        isCapturing = false;
        document.removeEventListener("keydown", handleKeyPress, true);
        document.removeEventListener("mousedown", handleMouseDown, true);
        document.removeEventListener("contextmenu", preventContext, true);
        document.removeEventListener("mouseup", preventNav, true);
        document.body.style.webkitAppRegion = '';
    };
    
    modal.querySelector("#hotkey-save-btn").addEventListener("click", async () => {
        if (!capturedKey || !keyType) {
            showNotification('error', 'No Key', 'Press a key or mouse button first');
            return;
        }
        
        cleanup();
        
        try {
            const result = await window.pywebview.api.macro_set_hotkey({
                type: keyType,
                value: capturedKey
            });
            
            if (result.success) {
                document.getElementById("keybind-display").textContent = result.hotkey;
                showNotification('success', 'Hotkey Set', result.message);
                overlay.remove();
            } else {
                showNotification('error', 'Failed', result.message);
            }
        } catch (err) {
            console.error(err);
            showNotification('error', 'Error', 'Failed to set hotkey');
        }
    });
    
    modal.querySelector("#hotkey-cancel-btn").addEventListener("click", () => {
        cleanup();
        overlay.remove();
    });
    
    overlay.addEventListener("click", (e) => {
        if (e.target === overlay) {
            cleanup();
            overlay.remove();
        }
    });
}

// --------- BUTTON HANDLERS ---------
function setupButtons() {
    // Add New button
    const addNewBtn = document.querySelector(".btn-primary");
    if (addNewBtn) {
        addNewBtn.addEventListener("click", showAddNewModal);
    }
    
    // All other buttons
    const buttons = document.querySelectorAll(".btn-ghost");
    buttons.forEach(btn => {
        const text = btn.textContent.trim();
        
        if (text === "Profiles") {
            btn.addEventListener("click", showProfilesModal);
        } else if (text === "Advanced Settings") {
            btn.addEventListener("click", showAdvancedSettingsModal);
        } else if (text === "Delete selected") {
            btn.addEventListener("click", async () => {
                const selected = document.querySelectorAll("#flags-table tbody tr.selected");
                if (selected.length === 0) {
                    showNotification('warning', 'No Selection', 'Please select flags to delete');
                    return;
                }
                
                try {
                    for (const row of selected) {
                        const flagName = row.querySelector(".flag-name").textContent;
                        await window.pywebview.api.delete_flag(flagName);
                    }
                    await loadFlags();
                    showNotification('success', 'Deleted', `Deleted ${selected.length} flag(s)`);
                } catch (err) {
                    console.error("Error deleting flags:", err);
                    showNotification('error', 'Delete Failed', 'Failed to delete selected flags');
                }
            });
        } else if (text === "Delete All") {
            btn.addEventListener("click", () => {
                showConfirmModal(
                    "Delete All Flags",
                    "Are you sure you want to delete all flags?",
                    async () => {
                        try {
                            await window.pywebview.api.delete_all_flags();
                            await loadFlags();
                            showNotification('warning', 'All Deleted', 'All flags have been removed');
                        } catch (err) {
                            console.error("Error deleting all flags:", err);
                            showNotification('error', 'Delete Failed', 'Failed to delete all flags');
                        }
                    }
                );
            });
        } else if (text === "Copy All") {
            btn.addEventListener("click", async () => {
                try {
                    const json = await window.pywebview.api.export_json();
                    navigator.clipboard.writeText(json);
                    showNotification('success', 'Copied', 'All flags copied to clipboard as JSON');
                } catch (err) {
                    console.error("Error copying flags:", err);
                    showNotification('error', 'Copy Failed', 'Failed to copy flags');
                }
            });
        } else if (text === "Export JSON") {
            btn.addEventListener("click", async () => {
                try {
                    const json = await window.pywebview.api.export_json();
                    const blob = new Blob([json], { type: 'application/json' });
                    const url = URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = 'flags.json';
                    a.click();
                    showNotification('success', 'Exported', 'Flags exported to JSON file');
                } catch (err) {
                    console.error("Error exporting JSON:", err);
                    showNotification('error', 'Export Failed', 'Failed to export JSON');
                }
            });
        } else if (text === "Clean FastFlag List") {
            btn.addEventListener("click", async () => {
                try {
                    await cleanInvalidFlags();
                } catch (err) {
                    console.error("Error cleaning flags:", err);
                    showNotification('error', 'Clean Failed', 'Failed to clean invalid flags');
                }
            });
        } else if (text === "Show Preset Flags") {
            btn.addEventListener("click", async () => {
                try {
                    const presets = await window.pywebview.api.get_preset_flags();
                    console.log("Available presets:", presets);
                    showNotification('info', 'Preset Flags', `${presets.length} preset flags available`);
                } catch (err) {
                    console.error("Error getting presets:", err);
                }
            });
        }
    });

    // Set Hotkey button in Clash Macro
    const setHotkeyBtn = document.getElementById("set-hotkey-btn");
    if (setHotkeyBtn) {
        setHotkeyBtn.addEventListener("click", showSetHotkeyModal);
    }
    
    // Bottom bar buttons
    const saveLaunchBtn = document.querySelector('.btn-bottom-primary');
    if (saveLaunchBtn) {
        saveLaunchBtn.addEventListener('click', async () => {
            try {
                // Clean invalid flags first
                const cleanResult = await window.pywebview.api.validate_and_clean_flags();
                if (cleanResult.removed_count > 0) {
                    await loadFlags();
                    showNotification('warning', 'Cleaned & Launching', 
                        `Removed ${cleanResult.removed_count} invalid flag(s), launching Roblox...`);
                } else {
                    showNotification('success', 'Launching', 'Applying flags and launching Roblox...');
                }
                
                await window.pywebview.api.save_and_launch();
            } catch (err) {
                console.error("Error launching:", err);
                showNotification('error', 'Launch Failed', 'Failed to launch Roblox');
            }
        });
    }
    
    const bottomButtons = document.querySelectorAll('.btn-bottom-secondary');
    bottomButtons.forEach(btn => {
        const text = btn.textContent.trim();
        
        if (text === 'Save') {
            btn.addEventListener('click', async () => {
                try {
                    const status = await window.pywebview.api.get_roblox_status();
                    
                    if (status === 'attached') {
                        // Clean invalid flags first
                        const cleanResult = await window.pywebview.api.validate_and_clean_flags();
                        if (cleanResult.removed_count > 0) {
                            await loadFlags();
                        }
                        
                        // Then apply
                        await window.pywebview.api.apply_flags_now();
                        showNotification('success', 'Saved & Applied', 
                            `Flags saved and applied to running Roblox`);
                    } else {
                        showNotification('success', 'Saved', 
                            'Flags saved - will apply when Roblox launches');
                    }
                } catch (err) {
                    console.error("Error saving:", err);
                    showNotification('error', 'Save Failed', 'Failed to save flags');
                }
            });
        } else if (text === 'Restart') {
            btn.addEventListener('click', async () => {
                try {
                    showNotification('info', 'Restarting', 'Closing and relaunching Roblox...');
                    await window.pywebview.api.restart_roblox();
                } catch (err) {
                    console.error("Error restarting:", err);
                    showNotification('error', 'Restart Failed', 'Failed to restart Roblox');
                }
            });
        } else if (text === 'Close') {
            btn.addEventListener('click', () => {
                if (window.pywebview) {
                    window.pywebview.api.close();
                }
            });
        }
    });
}

function initApp() {
    setupSidebar();
    setupNavigation();
    setupWindowButtons();
    setupSearch();
    setupButtons();
    setupClashMacroSlider();

    loadFlags();

    }
